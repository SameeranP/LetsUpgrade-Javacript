Various methods in console

Press f12 and open the console to view the assignment
console.time("your code took:")// Initialize timing the codeblock
console.log('Hello this is your console') //This is used to write something in the console
console.table({name : "sam", age: "20", marks: "100" }) //Used to view data in tabular form
console.warn("This is a warning. I am warning you!!") //Used to give a warning
console.timeEnd("your code took:")// display the time it took to run this code block
var age=190
console.assert(age<100, 'Age >100 not possible') //Used to assert if the given statement is true or not
console.error("This is an error") //To give user an error message
console.clear() // To clear the console
